BashのArrayについて


1. Arrayとは

一次元のリストのようなものです。各要素はwhitespaceで分けられています。新しくArrayを作成する場合は次のようにします。

arr=("a" "b c" "d" "e" "f" "g" "h")

2. データへのアクセス

#${array}では最初の要素のみにアクセスします
$ echo "${arr}"
a

#すべての要素にアクセスする時は${arr[@]}とします
 echo "${arr[@]}"
a b c d e f g h

#特定の要素にアクセスする時は${arr[n]}とします
$ echo "${arr[3]}"
e

# n番目からm個の要素にアクセスする時は${arr[@]:n-1:m}とします
$ echo "${arr[@]:1:2}"
b c d

# 最後から数えてn番目以降の要素にアクセスする時は${arr[@]: -n}とします ※:のあとの空白は必須です
$ echo "${arr[@]: -2}"
g h

# (文字の場合) ${arr[@],,}で小文字にすることができます
$ echo "${arr[@],,}"
a b c d e f g h

# (文字の場合)${arr[@]^}で各要素の先頭1文字を大文字にすることができます
$ echo "${arr[@]^}"
A B c D E F G H

#　(文字の場合)${arr[@]^}で大文字にすることができます
$ echo "${arr[@]^^}"
A B C D E F G H

# ${#arr[@]}で要素の数を取得できます
$ echo "${#arr[@]}"
7
